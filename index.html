    <script>

        function estimateFinishTime() {

            var startTimeInput = document.getElementById("start-time").value;

            var currentHoleInput = document.getElementById("current-hole").value;

            var resultContainer = document.getElementById("result");


            var dateFormatter = new Intl.DateTimeFormat("en-US", {

                hour: "numeric",

                minute: "numeric",

                hour12: false

            });


            var startDate = dateFormatter.format(new Date(startTimeInput));


            if (!startDate || startDate === "Invalid Date") {

                resultContainer.textContent = "Invalid start time format.";

                return;

            }


            var hole = parseInt(currentHoleInput);


            if (isNaN(hole) || hole < 1 || hole > 18) {

                resultContainer.textContent = "Invalid current hole number.";

                return;

            }


            var timePerHoleInSeconds = (60 * 60) / 18;

            var finishTime = new Date(new Date(startDate).getTime() + timePerHoleInSeconds * hole * 1000);

            var finishTimeString = dateFormatter.format(finishTime);


            resultContainer.textContent = "Estimated Finish Time: " + finishTimeString;

        }

    </script>
